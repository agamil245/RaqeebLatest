cpp
// 1. التعريفات الأساسية (يجب أن تكون في البداية)
#define TINY_GSM_MODEM_SIM7600  // استخدم 7600 إذا لم ينجح 7670 في التجميع
#define TINY_GSM_RX_BUFFER 1024

#include <TinyGsmClient.h>

// 2. تعريف دبابيس اللوحة (T-SIM7670G S3)
const int RX_PIN = 44;
const int TX_PIN = 43;
const int PWR_PIN = 4;   // دبوس تشغيل المودم
const int IND_PIN = 12;  // LED الحالة

// 3. إعداد المودم والسيريال
HardwareSerial SerialAT(1);
TinyGsm modem(SerialAT);

void setup() {
  Serial.begin(115200);
  delay(1000);

  pinMode(PWR_PIN, OUTPUT);
  pinMode(IND_PIN, OUTPUT);

  // تشغيل المودم (نبضة تشغيل)
  Serial.println("Wait...");
  digitalWrite(PWR_PIN, LOW);
  delay(100);
  digitalWrite(PWR_PIN, HIGH);
  delay(500);
  digitalWrite(PWR_PIN, LOW);

  // بدء التواصل مع شريحة SIM7670G
  SerialAT.begin(115200, SERIAL_8N1, RX_PIN, TX_PIN);
  delay(3000);

  Serial.println("Initializing modem...");
  if (!modem.init()) {
    Serial.println("Failed to restart modem, check power/wiring");
    return;
  }

  // تفعيل الـ GPS (أهم خطوة)
  Serial.println("Enabling GPS...");
  modem.sendAT("+CGNSSPWR=1"); // تشغيل الطاقة لوحدة GNSS
  modem.waitResponse();
  
  modem.enableGPS(); 
  Serial.println("GPS Started. Go outside and wait 2-5 minutes.");
}

void loop() {
  float lat, lon, speed, alt;
  int vsat, usat;
  float accuracy;
  int year, month, day, hour, minute, second;

  // محاولة الحصول على إحداثيات الموقع
  if (modem.getGPS(&lat, &lon, &speed, &alt, &vsat, &usat, &accuracy,
                   &year, &month, &day, &hour, &minute, &second)) {
    
    Serial.println("\n--- GPS DATA ---");
    Serial.print("Latitude: "); Serial.println(lat, 6);
    Serial.print("Longitude: "); Serial.println(lon, 6);
    Serial.print("Satellites: "); Serial.println(usat);
    Serial.print("Time: "); Serial.print(hour); Serial.print(":"); Serial.println(minute);
    Serial.println("----------------\n");
    digitalWrite(IND_PIN, HIGH); 
  } else {
    // إذا لم يجد إشارة، يطبع نقطة كل ثانيتين
    Serial.print(".");
    digitalWrite(IND_PIN, !digitalRead(IND_PIN)); // وميض LED للبحث
  }
  
  delay(2000);
}

cpp
// 1. التعريفات الأساسية (يجب أن تكون في البداية)
#define TINY_GSM_MODEM_SIM7670 
#define TINY_GSM_RX_BUFFER 1024

#include <TinyGsmClient.h>

// 2. دبابيس لوحة T-SIM7670G S3 (هذه الدبابيس ثابتة لهذا الموديل)
#define PIN_RX      44
#define PIN_TX      43
#define PIN_PWR     4   // دبوس الطاقة للمودم
#define PIN_LED     12  // LED الحالة

HardwareSerial SerialAT(1);
TinyGsm modem(SerialAT);

void setup() {
  Serial.begin(115200);
  delay(2000);
  Serial.println("\n--- بدء اختبار LILYGO T-SIM7670G S3 ---");

  pinMode(PIN_LED, OUTPUT);
  pinMode(PIN_PWR, OUTPUT);

  // 3. تسلسل تشغيل المودم (Power-On Sequence)
  // المودم يحتاج نبضة HIGH ثم LOW للبدء
  Serial.println("جاري محاولة تشغيل المودم (Powering ON)...");
  digitalWrite(PIN_PWR, LOW);
  delay(100);
  digitalWrite(PIN_PWR, HIGH);
  delay(1000); // نبضة تشغيل لمدة ثانية
  digitalWrite(PIN_PWR, LOW);
  
  // انتظر 5 ثوانٍ ليقلع المودم بالكامل
  delay(5000);

  // 4. بدء التواصل عبر السيريال
  SerialAT.begin(115200, SERIAL_8N1, PIN_RX, PIN_TX);
  delay(2000);

  // 5. فحص التواصل مع المودم
  Serial.println("Checking communication with modem...");
  int retry = 0;
  while (!modem.testAT() && retry < 10) {
    Serial.print("Attempt "); Serial.println(retry);
    delay(1000);
    retry++;
  }

  if (retry >= 10) {
    Serial.println("خطأ: المودم لا يستجيب! تأكد من كابل الـ USB أو البطارية.");
    return;
  }

  Serial.println("Modem Ready!");

  // 6. تفعيل الـ GPS
  Serial.println("Enabling GPS Power...");
  modem.sendAT("+CGNSSPWR=1"); // تشغيل الطاقة لوحدة الـ GPS
  modem.waitResponse();
  
  modem.enableGPS(); 
  Serial.println("GPS is active. Searching for satellites...");
}

void loop() {
  float lat, lon;
  
  // محاولة جلب الإحداثيات
  if (modem.getGPS(&lat, &lon)) {
    Serial.printf("\nLatitude: %.6f, Longitude: %.6f\n", lat, lon);
    digitalWrite(PIN_LED, HIGH);
  } else {
    // طباعة الحالة الحالية للأقمار (Diagnostic)
    Serial.print("Waiting for Fix... Status: ");
    modem.sendAT("+CGNSSINFO");
    String res = modem.readResponseUntil('\n');
    Serial.println(res);
    
    digitalWrite(PIN_LED, !digitalRead(PIN_LED)); // وميض للبحث
  }
  
  delay(5000);
}

لماذا قد "لا يحدث شيء"؟ (قائمة الفحص):
سرعة الـ Serial Monitor: تأكد أنك اخترت 115200 baud في نافذة Arduino Serial Monitor.
مصدر الطاقة: المودم يسحب تياراً عالياً (Peak Current) لحظة التشغيل. إذا كنت تستخدم منفذ USB ضعيف في الكمبيوتر، قد لا يقلع المودم. جرب توصيل بطارية Li-po أو منفذ USB أقوى.
ترتيب الـ Pins: تأكد أنك لم تقم بعكس RX و TX في الكود (القيم 43 و 44 صحيحة لـ S3).
تحديث المكتبة: إذا لم يظهر "Modem Ready"، فالمشكلة في المكتبة. تأكد أنك حذفت نسخة TinyGSM القديمة تماماً واستخدمت هذه النسخة.
بعد رفع هذا الكود، ماذا يظهر لك في الـ Serial Monitor بالضبط؟ هل يطبع "Checking communication"؟

// دبابيس T-SIM7670G S3
const int RX_PIN = 44;
const int TX_PIN = 43;
const int PWR_PIN = 4;

HardwareSerial SerialAT(1);
TinyGsm modem(SerialAT);

void setup() {
  Serial.begin(115200);
  delay(1000);

  pinMode(PWR_PIN, OUTPUT);

  // تشغيل المودم
  Serial.println("Powering ON...");
  digitalWrite(PWR_PIN, LOW);
  delay(100);
  digitalWrite(PWR_PIN, HIGH);
  delay(1000);
  digitalWrite(PWR_PIN, LOW);
  delay(3000);

  SerialAT.begin(115200, SERIAL_8N1, RX_PIN, TX_PIN);
  delay(3000);

  if (!modem.init()) {
    Serial.println("Modem Init Failed");
    return;
  }

  // تفعيل الـ GPS
  Serial.println("Enabling GPS...");
  modem.sendAT("+CGNSSPWR=1"); 
  modem.waitResponse();
  modem.enableGPS();
}

void loop() {
  float lat, lon;
  if (modem.getGPS(&lat, &lon)) {
    Serial.printf("LAT: %f, LON: %f\n", lat, lon);
  } else {
    // طباعة الحالة للتأكد من وجود طاقة
    Serial.print("Searching... ");
    modem.sendAT("+CGNSSINFO");
    modem.waitResponse(1000);
  }
  delay(5000);
}
